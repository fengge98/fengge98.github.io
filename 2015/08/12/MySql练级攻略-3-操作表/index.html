MySql练级攻略(3)-操作表

  • mysql

记录表相关的增删改查操作包括：建库,表操作，记录的操作

database

mysql> create database test;
mysql> show databases;
mysql> drop database test;//数据库的创建删除

table

建表

mysql> create table admin (
mysql> id           int      not null primary key auto_increment,
mysql> username  varchar(30) not null             default 'test',  
mysql> sex      enum('F','M'),
mysql> address   var char(50)  null);

查看

mysql> desc admin;

修改

mysql> alter table admin add tel1 char(11) not null ; //添加字段
mysql> alter table admin add tel2 after sex ; //指定位置添加字段
mysql> alter table admin drop tel2 ;//删除字段
mysql> alter table admin change tel1 blog varchar(30) null ;//将tel1改为blog后面至少要有修改后的字段类型.
mysql> alter table admin change id id int after username;//调整字段位置
mysql> alter table admin renanme admin5;//重命名表

以后用到的语句再添加

删除

mysql> drop table admin;

record

select id,username from admin where 1=1 ;
insert into admin (username,tel1) values("admin","13000000000");//除自增和带有默认值的字段外，NOT NULL的字段必须要进行赋值.
delete from admin where username = "admin";//更新
