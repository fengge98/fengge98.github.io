SQL注入练级攻略(4)-数据库常量及函数

发表于 2015-08-12   |   分类于 SLQ注入练级系列   |  

常量

version

VERSION()
@@VERSION
@@GLOBAL.VERSION
mysql> select @@version;//显示mysql版本
mysql> select * from user where 1 and mid(version(),1,1) = 5;//当数据库版本为5.x.xx时返回为真

user

user()
current_user
current_user()
system_user()
session_user()
mysql> select user(); //显示当前用户
mysql> select concat_ws(":",user,password) from mysql.user where user = "root"; //显示 root:"对应用户密码散列值"

database

database()
schema()
mysql> select database();//显示mysql
select schema_name from information_schema.schemata;//显示所有数据库

hostname

@@hostname
mysql> select @@hostname;

函数

length

length()
mysql> select length("abc");//字符串长度

mid

mid()
mysql> select mid("abc",1,1);//指定字符串

if,case

if()
mysql> select if(length(user())>10,sleep(3),0);//if成立返回sleep(3)
case
mysql select case when 1=1 then true else false end;
//if相似

ascii

ascii()
mysql> select ascii("abcb")//返回a的ascii码97
mysql> select ascii(mid("abcb",2,1));//返回b的ascii码98

count

count()
mysql> select count(*) from mysql.user;//返回记录条数

concat,concat_ws

concat()
mysql> select concat("a","b");//显示ab

concat_ws()
mysql> select concat_ws(":",user,password) from mysql.user where user = "root"; //显示 root:"对应用户密码散列值"

rand

rand()
mysql> select rand();//0~1之间的随机数

floor

floor()
mysql> select floor(1.1);//向下取整

group_by

group by()
mysql> select type,max(price) from price_info group by type;
//先按tpye分组，建立若干临时表，然后在每一个临时表示执行max()函数

distinct

distinct()
mysql> select distinct user from mysql.user;
//排重，只显示不重复的user.

以上内容使用的数据库版本为mysql 5.6.17.
